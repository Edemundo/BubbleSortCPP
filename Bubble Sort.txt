//Declarações do programa de linguagem de programação

//Definições do programa
#define PAUSA						system("pause")

#include "B2.h"

//entry point do programa
void main(void)
{
	//Declaração de variáveis
	int v[3];
	int qInt=3;
	int k;
	int aux;
	int i;
	int j;
	k=qInt-1;

	//Inserindo as variáveis no vetor
	for(i=0;i<3;i++)
	{
		cout<<"Digite a "<<i+1<<" double: ";
		cin>>v[i];
		cout<<endl;
	}

	

	//Ordenando as variáveis no vetor
	for(i=0;i<qInt;i++)
	{
		for(j=0;j<k;j++)
		{
			if(v[j]>v[j+1])
			{
				aux=v[j];
				v[j]=v[j+1];
				v[j+1]=aux;
			}
		}
		k--;
	}

	//Exibindo o vetor ordenado
	for(i=0;i<3;i++)
	{
		cout<<v[i]<<" ";
	}

	PAUSA;
}
